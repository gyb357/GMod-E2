@name Afghanit APS ATGM Support V1
@inputs  [Sensor]:entity [Active]
@outputs [ATGM]:entity [CpuUsage Ops] Find
@persist [FindDist FindDelay FindAngle FindExcludeOwnProp]
@persist [Own]:entity [Range] [ClassAry]:array
@persist 
interval(64)

if (first() | dupefinished()) {
    # ----- User Interface ----- #
    FindDist  = 20
    FindDelay = 0.5
    FindAngle = 50
    
    FindExcludeOwnProp = 0
    
    # ----- Do Not Touch There ----- #
        Own      = owner()
        Range    = 50000
        ClassAry = array("acf_missile", "acf_glatgm")
    
    function entity findMissile() {
        if (FindExcludeOwnProp) {
            findExcludePlayerProps(Own)
        }
        findIncludeClass(ClassAry[1, string])
        findIncludeClass(ClassAry[2, string])
        findSortByDistance(Sensor:pos())
        findInCone(Sensor:pos(), Sensor:forward(), FindDist * 100, FindAngle)
        Ary  = findToArray()
        ATGM = Ary[1, entity]
        
        return ATGM
    }
    
    function number findDelay() {
        if (FindDelay != 0) {
            timer("Delay", FindDelay * 1000)
            
            if (clk("Delay")) {
                Find = 1
            } else {
                Find = 0
            }
        }
        
        return Find
    }
}

if (Active) {
    findDelay()
    if (Find) {
        findMissile()
    }
}

    CpuUsage = cpuUsage() * 1000000
    Ops      = ops()
