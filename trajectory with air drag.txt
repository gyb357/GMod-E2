#[
  G = 600 # 11.43m/s^2
  ToMeter = 39.37
  ToReal = (2 / 3)
  
  V0 = Gun:acfMuzzleVel() * ToMeter

  Pos = Ranger:pos()
  Dx = sqrt((Pos:x() - Gun:massCenter():x())^2 + (Pos:y() - Gun:massCetner():y())^2)
  Dx = Pos:z() - Gun:massCenter():z()
  
  Tangent = (-(V0^2) + sqrt(V0^4 - G^2 * Dx^2 + 2 * Dy * G * V0^2)) / (G * Dx)
  
  Theta = atan(Tangent)
  Pitch = Gun:angles():pitch()
  
  Fraera = pi() * (Gun:acfCaliber() / 20)^2
  DragCoef = Fraera / 10000 / Gun:acfProjectileMass()
  
  ToF = Dx / (V0 * cos(Pitch))
  
  Drag = vec(0, 0, ((0.5 * G * ((V0 / ToMeter) * ToReal) * ToF^3) / cos(Theta * 0.5)) + (Dx * DragCoef) * -sin(Theta) * ToF^3)
  Drop = vec(0, 0, (Gun:massCenter() + vec(Pos:distance(Gun:massCenter()), 0, 0):rotate(ang(Theta, 0, 0))):z()) - vec(0, 0, Pos:z())
  
  Fall = Drag + Drop
